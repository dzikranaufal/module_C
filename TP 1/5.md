SELESAI
FUNCTION hitungTotalDenda(dataHari)
    total_denda = 0
    FOR i = 0 TO 4 DO
        IF dataHari[i] > 0 THEN
            total_denda = total_denda + (dataHari[i] * 1000)
        ENDIF
    ENDFOR
    RETURN total_denda
END FUNCTION
PROCEDURE tampilData(dataHari)
    OUTPUT "----- Laporan Data -----"
    OUTPUT semua isi array dataHari
    total = hitungTotalDenda(dataHari)
    OUTPUT "Total Potensi Denda : ", total
END PROCEDURE
PROCEDURE bayarDenda(dataHari)
    OUTPUT "----- Pembayaran Denda -----"
    OUTPUT "Data Keterlambatan : "
    OUTPUT semua isi array dataHari
    INPUT index
    dataHari[index] = 0
    OUTPUT "Mahasiswa pada index tersebut telah lunas"
END PROCEDURE

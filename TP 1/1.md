Program 1.c

DEKLARASI
    kata : array string
    kata_baru : array string
    panjang : integer
    i : integer

Algoritma
MULAI
    INPUT kata
    SET panjang_string(kata)

    IF panjang < 3 THEN
        OUTPUT "Jumlah Karakter tidak boleh kurang dari 3"

    ELSE IF panjang > 15 THEN
        OUTPUT "Jumlah Karakter tidak boleh lebih dari 15"

    ELSE
        FOR i = 0 TO panjang - 1 :
            IF kata[i] = 'a' ATAU 'A' THEN
                kata_baru[i] = '1'
            ELSE IF kata[i] = 'e' ATAU 'E' THEN
                kata_baru[i] = '2'
            ELSE IF kata[i] = 'i' ATAU 'I' THEN
                kata_baru[i] = '3'
            ELSE IF kata[i] = 'o' ATAU 'O' THEN
                kata_baru[i] = '4'
            ELSE IF kata[i] = 'u' ATAU 'U' THEN
                kata_baru[i] = '5'
            ELSE
                kata_baru[i] = kata[i]
            ENDIF
        OUTPUT "Dari ", kata, ", menjadi ", kata_baru
    ENDIF

SELESAI

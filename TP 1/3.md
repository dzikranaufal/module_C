DEKLARASI
    lele : integer
    nila : integer
    gurame : integer
    hasil_lele : integer
    hasil_nila : integer
    hasil_gurame : integer

Algoritma
MULAI
    INPUT lele, nila, gurame
    OUTPUT "Bulan ke-1"
    OUTPUT "Lele : ", lele
    OUTPUT "Nila : ", nila
    OUTPUT "Gurame : ", gurame

    hasil_lele  = hitungLele(lele)
    hasil_nila  = hitungNila(nila, lele)
    hasil_gurame  = hitungGurame(gurame, hasil_nila)

    OUTPUT "Bulan ke-2"
    OUTPUT "Lele : ", hasil_lele
    OUTPUT "Nila : ", hasil_nila
    OUTPUT "Gurame : ", hasil_gurame

SELESAI
FUNCTION hitungLele(lele : integer)
    lele = lele * 3
    RETURN lele
END FUNCTION
FUNCTION hitungNila(nila : integer, lele : integer)
    nila = (nila - 10) + (lele / 2)
    RETURN nila
END FUNCTION
FUNCTION hitungGurame(gurame : integer, nila : integer)
    gurame = gurame + (nila / 3)
    RETURN gurame
END FUNCTION

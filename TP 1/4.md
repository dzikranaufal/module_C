Program 4.c

DEKLARASI
    kursi : array[10] integer
    nomor_kursi : integer
    pilihan : character
    status : boolean
    i : integer

Algoritma
MULAI
    SET kursi[0] sampai kursi[9] = 0
    SET pilihan = 'y'

    WHILE pilihan == 'y' DO
        OUTPUT "Status Kursi [0 = Kosong, 1 = Terisi] :"
        FOR i = 0 TO 9 DO
            OUTPUT kursi[i]
        END FOR

        OUTPUT "Masukkan Nomor Kursi yang ingin dipesan (1-10): "
        INPUT nomor_kursi
        status = cekKursi(kursi, nomor_kursi)
        IF status == true THEN
            CALL pesanKursi(kursi, nomor_kursi)
            OUTPUT "Booking Berhasil untuk kursi No ", nomor_kursi
        ELSE
            OUTPUT "Maaf, kursi nomor ", nomor_kursi, " sudah dipesan"
        END IF
        OUTPUT "Ingin memesan lagi? (y/n): "
        INPUT pilihan
    END WHILE
SELESAI
PROCEDURE pesanKursi(kursi : array integer, nomor_kursi : integer)
    kursi[nomor_kursi - 1] = 1
END PROCEDURE
FUNCTION cekKursi(kursi : array integer, nomor_kursi : integer)
    RETURN kursi[nomor_kursi - 1] == 0
END FUNCTION

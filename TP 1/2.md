Program 2.c

DEKLARASI
    jenis_buku : integer
    harga_buku : array [jenis_buku] integer
    jumlah_terjual : array [jenis_buku] integer
    total_harga : integer
    i : integer

Algoritma
MULAI
    INPUT jenis_buku
    SET total_harga = 0
    inputBuku(jenis_buku, harga_buku, jumlah_terjual)
    hitungTotal(jenis_buku, harga_buku, jumlah_terjual, total_harga)
    tampilkanLaporan(jenis_buku, harga_buku, jumlah_terjual, total_harga)
SELESAI

PROCEDURE inputBuku(jenis_buku : integer, harga_buku : array integer, jumlah_terjual : array integer)
    FOR i = 0 TO jenis_buku - 1
    OUTPUT "Masukkan harga buku ke-", i + 1
    INPUT harga_buku[i]
    OUTPUT "Masukkan jumlah buku terjual ke-", i + 1
    INPUT jumlah_terjual[i]
    END FOR
END PROCEDURE

PROCEDURE hitungTotal(jenis_buku : integer, harga_buku : array integer, jumlah_terjual : array integer, total_harga : integer)
    FOR i = 0 TO jenis_buku - 1
        total_harga ← total_harga +
                       (harga_buku[i] × jumlah_terjual[i])
    END FOR
END PROCEDURE

PROCEDURE tampilkanLaporan(jenis_buku : integer, harga_buku : array integer, jumlah_terjual : array integer, total_harga : integer)
    rata_rata : integer
    SET rata_rata = 0
    FOR i = 0 TO jenis_buku - 1
        rata_rata = rata_rata + harga_buku[i]
    END FOR
    rata_rata = rata_rata / jenis_buku

    OUTPUT "===== Laporan Penjualan ====="
    OUTPUT "Total Penjualan : ", total_harga
    OUTPUT "Rata-rata Harga Buku : ", rata_rata
END PROCEDURE
